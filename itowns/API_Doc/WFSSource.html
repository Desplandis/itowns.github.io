Add color layer with wfs source

const colorlayer = new ColorLayer('color_build', {
    style: {
        fill: 'red',
        fillOpacity: 0.5,
        stroke: 'white',
    },
    source: {
       url: 'http://wxs.fr/wfs',
       protocol: 'wfs',
       version: '2.0.0',
       typeName: 'BDTOPO_BDD_WLD_WGS84G:bati_remarquable',
       projection: 'EPSG:4326',
       extent: {
           west: 4.568,
           east: 5.18,
           south: 45.437,
           north: 46.03,
       },
       format: 'application/json',
    }
});
// Add the layer
view.addLayer(colorlayer);

Add geometry layer with wfs source

const geometrylayer = new GeometryLayer('mesh_build', {
    update: itowns.FeatureProcessing.update,
    convert: itowns.Feature2Mesh.convert({ extrude: () => 50 }),
    source: {
       protocol: 'wfs',
       url: 'http://wxs.fr/wfs',
       version: '2.0.0',
       typeName: 'BDTOPO_BDD_WLD_WGS84G:bati_remarquable',
       projection: 'EPSG:4326',
       extent: {
           west: 4.568,
           east: 5.18,
           south: 45.437,
           north: 46.03,
       },
       zoom: { min: 14, max: 14 },
       format: 'application/json',
    }
});
// Add the layer
view.addLayer(geometrylayer);
